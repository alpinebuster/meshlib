202template <typename T>
203bool checkboxFlags( const char* label, T& target, T flags )
204{
205 bool value = bool( target & flags );
206 bool mixed = value && ( target & flags ) != flags;
207 if ( checkboxMixed( label, &value, mixed ) )
208 {
209 if ( value )
210 target |= flags;
211 else
212 target &= ~flags;
213 return true;
214 }
215 return false;
216}

311 // Whether `T` is a scalar type that we can use with our widgets.
312 template <typename T>
313 concept Scalar = std::is_arithmetic_v<T> && !std::is_same_v<T, bool>;
314
315 // Whether `T` is a scalar or vector that we can use with our widgets.
316 template <typename T>
317 concept VectorOrScalar = Scalar<typename VectorTraits<T>::BaseType>;
318
319 // Whether `Bound` is a valid min/max bound for `Target`.
320 // That is, either the same type, or if `Target` is a vector, `Bound` can also be a scalar of the same type.
321 template <typename Bound, typename Target>
322 concept ValidBoundForTargetType =
323 std::same_as<Bound, Target> ||
324 ( VectorTraits<Bound>::size == 1 && std::same_as<typename VectorTraits<Bound>::BaseType, typename VectorTraits<Target>::BaseType> );
325
326 // Whether `Speed` is a valid drag speed type for `Target`.
327 // That is, either a single/vector of `float` or the same type as target (or its element if it's a vector).
328 template <typename Speed, typename Target>
329 concept ValidDragSpeedForTargetType =
330 std::same_as<Speed, typename VectorTraits<Target>::BaseType> || std::same_as<Speed, float> ||
331 std::same_as<Speed, Target> || std::same_as<Speed, typename VectorTraits<Target>::template ChangeBase<float>>;

333 // A common code for sliders and other widgets dealing with measurement units.
334 // `E` must be explicitly set to a measurement unit enum. The other template parameters are deduced.
335 // `label` is the widget label, `v` is the target value.
336 // `func` draws the widget for an individual scalar. We call it more than once for vectors.
337 // `func` is `( const char* label, auto& elem, int i ) -> bool`.
338 // It receives `elem` already converted to the display units (so you must convert min/max bounds manually). `i` is the element index for vectors.
339 // When `v` is integral, `func` will be instantiated for both integral and floating-point element type. The latter is required if we're doing conversions.
340 // NOTE: For integral `v`, in `func` you must look at the type of `elem` and convert your min/max bounds (etc) to the same type.
341 // Notice `unitParams` being accepted by an lvalue reference. For convenience, we reset the `sourceUnit` in it before calling the user callback,
342 // since at that point no further conversions are necessary.
343 template <UnitEnum E, VectorOrScalar T, typename F>
344 [[nodiscard]] bool unitWidget( const char* label, T& v, UnitToStringParams<E>& unitParams, F&& func );
