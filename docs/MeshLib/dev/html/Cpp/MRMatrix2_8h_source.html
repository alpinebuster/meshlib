147template <typename T>

149{
150 const auto x = cross( from, to );
151 if ( x > 0 )
152 return rotation( angle( from, to ) );
153 if ( x < 0 )
154 return rotation( -angle( from, to ) );
155 if ( dot( from, to ) >= 0 )

157 return rotation( T( PI ) );
