589void MeshTopology::forEachVertex( const MeshTriPoint & p, T && callback ) const
590{
591 if ( auto v = p.inVertex( *this ) )
592 {
593 callback( v );

609template<typename T>
610void MeshTopology::flipEdgesIn( const EdgeId e0, T && flipNeeded )
611{
612 EdgeId e = e0;
613 for (;;)
614 {
615 auto testEdge = prev( e.sym() );
616 if ( left( testEdge ) && right( testEdge ) && flipNeeded( testEdge ) )
617 flipEdge( testEdge );
618 else
619 {
620 e = next( e );
621 if ( e == e0 )
622 break; // full ring has been inspected
623 }
624 }
625}

627template<typename T>
628void MeshTopology::flipEdgesOut( EdgeId e0, T && flipNeeded )
629{
630 EdgeId e = e0;
631 for (;;)
632 {
633 if ( left( e ) && right( e ) && flipNeeded( e ) )
634 {
635 e0 = next( e );
636 flipEdge( e );
637 e = e0;
638 }
639 else
640 {
641 e = next( e );
642 if ( e == e0 )
643 break; // full ring has been inspected
644 }
645 }

648// rearrange vector values by map (old.id -> new.id)
649template<typename T, typename I>
650[[nodiscard]] Vector<T, I> rearrangeVectorByMap( const Vector<T, I>& oldVector, const BMap<I, I>& map )
651{
652 Vector<T, I> newVector;
653 newVector.resize( map.tsize );
654
655 const auto& mData = map.b.data();
656 const auto sz = std::min( oldVector.size(), map.b.size() );
657 for ( I i = I(0); i < sz; ++i)
658 {
659 I newV = mData[i];
660 if ( newV.valid() )
661 newVector[newV] = oldVector[i];
662 }
663 return newVector;
664}
