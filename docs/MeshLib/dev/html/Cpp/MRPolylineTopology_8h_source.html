178template<typename T, typename F1, typename F2>
179void PolylineTopology::buildFromContours( const std::vector<std::vector<T>> & contours, F1 && reservePoints, F2 && addPoint )
180{
181 *this = {};
182 size_t size = 0;
183 std::vector<bool> closed;
184 closed.reserve( contours.size() );
185 int numClosed = 0;
186 for ( const auto& c : contours )
187 {
188 const auto csize = c.size();
189 if ( csize > 2 )
190 {
191 closed.push_back( c.front() == c.back() );
192 }
193 else
194 {
195 closed.push_back( false );
196 }
197 if ( csize < 2 )
198 continue; // ignore contours with 0 or 1 points because of no edges in them
199 size += c.size();
200 if ( closed.back() )
201 ++numClosed;
202 }
203
204 reservePoints( size - numClosed );
205 vertResize( size - numClosed );
206
207 for ( int i = 0; i < contours.size(); ++i )
208 {
209 const auto& c = contours[i];
210 if ( c.size() < 2 )
211 continue; // ignore contours with 0 or 1 points because of no edges in them
212 const auto e0 = makeEdge();
213 const auto v0 = addPoint( c[0] );
214 setOrg( e0, v0 );
215 auto e = e0;
216 for ( int j = 1; j + 1 < c.size(); ++j )
217 {
218 const auto ej = makeEdge();
219 splice( ej, e.sym() );
220 const auto vj = addPoint( c[j] );
221 setOrg( ej, vj );
222 e = ej;
223 }
224 if ( closed[i] )
225 {
226 splice( e0, e.sym() );
227 }
228 else
229 {
230 const auto vx = addPoint( c.back() );
231 setOrg( e.sym(), vx );
232 }
233 }
234 assert( isConsistentlyOriented() );
235 assert( edgePerVertex_.size() == size - numClosed );
236}

238template<typename T, typename F>
239std::vector<std::vector<T>> PolylineTopology::convertToContours( F&& getPoint, std::vector<std::vector<VertId>>* vertMap ) const
240{
241 std::vector<std::vector<T>> res;
242
243 UndirectedEdgeBitSet linesUsed;
244 linesUsed.autoResizeSet( UndirectedEdgeId{ undirectedEdgeSize() } );
245 linesUsed.flip();
246 for ( EdgeId e0 : linesUsed )
247 {
248 if ( isLoneEdge( e0 ) )
249 continue;
250
251 EdgeId curLine = e0;
252 while ( curLine != next( curLine ) )
253 {
254 curLine = next( curLine ).sym();
255 if ( curLine == e0 )
256 break;
257 }
258
259 linesUsed.set( curLine.undirected(), false );
260
261 EdgeId e = curLine;
262 std::vector<T> cont;
263 std::vector<VertId> map;
264 auto orgV = org( e );
265 cont.push_back( getPoint( orgV ) );
266 if ( vertMap )
267 map.push_back( orgV );
268 for ( ;; )
269 {
270 e = e.sym();
271 cont.push_back( getPoint( org( e ) ) );
272 e = next( e );
273 if ( !linesUsed.test_set( e.undirected(), false ) )
274 break;
275 }
276 res.push_back( std::move( cont ) );
277 if ( vertMap )
278 vertMap->push_back( std::move( map ) );
279 }
280
281 return res;
282}

285struct PolylineMaker
286{
287 PolylineTopology & topology;
288 PolylineMaker( PolylineTopology & t ) : topology( t ) {}
289
292 EdgeId start( VertId v )
293 {
294 assert( !e0_ && !eLast_ );
295 e0_ = eLast_ = topology.makeEdge();
296 topology.setOrg( e0_, v );
297 return e0_;
298 }
301 EdgeId proceed( VertId v )
302 {
303 assert( eLast_ );
304 const auto ej = topology.makeEdge();
305 topology.splice( ej, eLast_.sym() );
306 topology.setOrg( ej, v );
307 return eLast_ = ej;
308 }
310 void close()
311 {
312 assert( e0_ && eLast_ );
313 topology.splice( e0_, eLast_.sym() );
314 e0_ = eLast_ = {};
315 }
317 void finishOpen( VertId v )
318 {
319 assert( eLast_ );
320 topology.setOrg( eLast_.sym(), v );
321 e0_ = eLast_ = {};
322 }
323
324private:
325 EdgeId e0_, eLast_;
326};
