33// If a type derived from `VisualObject` wants to extend `VisualizeMaskType`, it must create a separate enum and specialize this to `true` for it.
34// NOTE! All those enums can start from 0, don't worry about collisions.
35template <typename T> struct IsVisualizeMaskEnum : std::false_type {};
36template <> struct IsVisualizeMaskEnum<VisualizeMaskType> : std::true_type {};
37
38// Wraps `IsVisualizeMaskEnum` and adds some sanity checks.
39template <typename T>
40concept AnyVisualizeMaskEnumType =
41 IsVisualizeMaskEnum<T>::value &&
42 std::is_same_v<std::underlying_type_t<T>, int> &&
43 std::is_same_v<T, std::remove_cvref_t<T>>;
44
45// Stores a `VisualizeMaskType` or any other enum that extends it (i.e. which specializes `IsVisualizeMaskEnum`).
46// To extract the value, do this:
47// if ( auto value = x.tryGet<MyEnum>() )
48// {
49// switch ( *value )
50// {
51// case MyEnum::foo: ...
52// case MyEnum::bar: ...
53// }
54// }
55// else // forward to the parent class
56class AnyVisualizeMaskEnum
57{
58 std::type_index type_;
59 int value_ = 0;
60
61public:
62 template <AnyVisualizeMaskEnumType T>
63 AnyVisualizeMaskEnum( T value ) : type_( typeid(T) ), value_( decltype(value_)( value ) ) {}
64
65 template <AnyVisualizeMaskEnumType T>
66 [[nodiscard]] std::optional<T> tryGet() const
67 {
68 if ( type_ == typeid(T) )
69 return T( value_ );
70 else
71 return {};
72 }
73};
