176Polyline<U> Polyline<V>::toPolyline() const
177{
178 Polyline<U> res;
179 res.topology = topology;
180 res.points.reserve( points.size() );
181 for ( size_t i = 0; i < points.size(); i++ )
182 {
183 res.points.push_back( U{ points[VertId( i )] } );
184 }
