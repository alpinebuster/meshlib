36[[nodiscard]] V closestPointOnLineSegm( const V& pt, const LineSegm<V> & l )
37{
38 auto ab = l.b - l.a;
39 auto dt = dot( pt - l.a, ab );
40 auto abLengthSq = ab.lengthSq();
41 if ( dt <= 0 )
42 return l.a;
43 if ( dt >= abLengthSq )
44 return l.b;
45 auto ratio = dt / abLengthSq;
46 return l.a * ( 1 - ratio ) + l.b * ratio;
47}
