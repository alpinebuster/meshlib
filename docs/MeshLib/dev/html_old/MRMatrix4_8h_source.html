184template <typename T>
185Matrix3<T> Matrix4<T>::submatrix3( int i, int j ) const noexcept
186{
187 Matrix3<T> res;
188 auto* resM = (T*) &res.x;
189 int cur = 0;
190 for ( int m = 0; m < 4; m++ )
191 {
192 if ( m == i )
193 continue;
194 for ( int n = 0; n < 4; n++ )
195 {
196 if ( n == j )
197 continue;
198 resM[cur++] = (*this)[m][n];
199 }
200 }
201 assert( cur == 9 );
202 return res;
203}
