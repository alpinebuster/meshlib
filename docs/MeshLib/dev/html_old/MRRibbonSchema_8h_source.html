151 {
152 const auto& items = RibbonSchemaHolder::schema().items;
153 for ( const auto& item : items )
154 {
155 auto plugin = std::dynamic_pointer_cast< T >( item.second.item );
156 if ( !plugin )
157 continue;
158 f( plugin_ = std::move( plugin ) );
159 break;
160 }
161 }

165 g_( plugin_ );
166 }
