224// overload `toString` functoion to throw exception from custom `Expected::error` type
225template<typename E>
226[[noreturn]] void throwExceptionFromExpected(const E& err)
227{
228 if constexpr (std::is_nothrow_convertible<E, std::string>::value)
229 throw std::runtime_error(err);
230 else
231 throw std::runtime_error(toString(err));
232}

234// Like `e.value()`, but throws using `throwExceptionFromExpected` (which is better, because it allows Python to see the proper error message), and also supports `T == void`.
235template <typename T>
236[[nodiscard]] decltype(auto) expectedValueOrThrow( T&& e )
237{
238 if ( e )
239 {
240 if constexpr ( std::is_void_v<typename std::remove_cvref_t<T>::value_type> )
241 return;
242 else
243 return *std::forward<T>( e );
244 }
245 else
246 {
247 throwExceptionFromExpected( e.error() );
248 }
249}

251template<typename R, typename E, typename... Args>
252auto decorateExpected( std::function<Expected<R, E>( Args... )>&& f ) -> std::function<R( Args... )>
253{
254 return[fLocal = std::move( f )]( Args&&... args ) mutable -> R
255 {
256 auto res = fLocal(std::forward<Args>( args )...);
257 if (!res.has_value())
258 throwExceptionFromExpected(res.error());
259
260 if constexpr (std::is_void<R>::value)
261 return;
262 else
263 return res.value();
264 };
265}

268auto decorateExpected( F&& f )
